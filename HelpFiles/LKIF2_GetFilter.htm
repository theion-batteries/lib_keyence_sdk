Get filter mode (LKIF2_GetFilter)
　 　 　
RC WINAPI LKIF2_GetFilter(IN int OutNo,OUT LKIF_FILTERMODE *FilterMode,OUT LKIF_FILTERPARA *FilterPara);
　
Description Obtain the filter mode and filter parameter.
　
Parameter OutNo  Specify the OUT No. The value range is 0 to ((the No. of used OUT)-1).
FilterMode  The filter mode is contained.
　
typedef enum {
LKIF_FILTERMODE_MOVING_AVERAGE, // Moving average
LKIF_FILTERMODE_LOWPASS, // Low-pass filter
LKIF_FILTERMODE_HIGHPASS, // High-pass filter
} LKIF_FILTERMODE;
FilterPara The filter parameter is contained.
(The number of averaging for moving average, or the cutoff frequency for LPF/HPF.)
　
typedef enum {
LKIF_FILTERPARA_AVE_1 = 0, // 1 time of averaging
LKIF_FILTERPARA_AVE_4, //
LKIF_FILTERPARA_AVE_16, //
LKIF_FILTERPARA_AVE_64, //
LKIF_FILTERPARA_AVE_256, //
LKIF_FILTERPARA_AVE_1024, //
LKIF_FILTERPARA_AVE_4096, //
LKIF_FILTERPARA_AVE_16384, //
LKIF_FILTERPARA_AVE_65536, //
LKIF_FILTERPARA_AVE_262144, // 262144 times of averaging
　
LKIF_FILTERPARA_COFF_3000 = 0, // 3000 Hz
LKIF_FILTERPARA_COFF_1000, // 1000 Hz
LKIF_FILTERPARA_COFF_300, // 300 Hz
LKIF_FILTERPARA_COFF_100, // 100 Hz
LKIF_FILTERPARA_COFF_30, // 30 Hz
LKIF_FILTERPARA_COFF_10, // 10 Hz
LKIF_FILTERPARA_COFF_3, // 3 Hz
LKIF_FILTERPARA_COFF_1, // 1 Hz
LKIF_FILTERPARA_COFF_0_3, // 0.3 Hz
LKIF_FILTERPARA_COFF_0_1, // 0.1 Hz
} LKIF_FILTERPARA;
